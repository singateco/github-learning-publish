    function a() {
    return this.num;

function plus(money) {
    this. num += money;
}

function minus(money) {
    this.num -= money;
}

var mon = new Object();
mon.name = "김길동";
mon.id = "123";
mon.num = 30000; //초기값 설정

mon.a = a; //메소드 작성
mon.plus = plus;
mon.minus = minus;

    document.write(mon.name);

    리터럴 표기법 객체¶

    <!DOCTYPE html>
<html lang="en">
<head>
  <script>
    //리터럴 표기법
    var n={
      name:"준형",
      id:"123",
      money:5000,

      a:function(){
        return this.money;
      },

      plus:function(num){
        this.money+=num;
      },

      minus:function(num){
        this.money-=num;
      }

    }
  </script>
</head>
<body>
  <script>

    n.plus(5000);
    document.write(n.a()+"<br>");

    n.minus(4000);
    document.write(n.a()+"<br>");


  </script>
</body>
</html>

    프로토타입으로 객체 만들기¶

    <!DOCTYPE html>
<html lang="en">
<head>
  <script>
    function mon(name,id,n){
      this.name=name;
      this.id=id;
      this.n=n;

      this.a=function(){
        return this.n;
      }
      this.plus=function(money){
        this.n+=money;
      }
      this.minus=function(money){
        this.n-=money;
      }
    }
  </script>
</head>
<body>
  <script>

   var aa=new mon("승주","123",30000); //생성자
   document.write(aa.name);

   aa.plus(10000);
   document.write(aa.a()+"<br>");

   aa.minus(10000);
   document.write(aa.a()+"<br>");

  </script>
</body>
</html>
