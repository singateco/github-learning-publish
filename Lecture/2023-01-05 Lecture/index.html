    appendChild¶

    <head>
    <script>
        function di() {
            var n = document.getElementById("Body");
            var m = document.createElement("div");
            m.innerHTML = "new div";
            m.setAttribute("id", "new");
            m.style.backgroundColor = "orange";
            n.appendChild(m);

            m.onclick = function() {
                var p = this.parentElement;
                p.removeChild(this);
            }
        }
    </script>
</head>
<body id="Body">
    <p>
        테스트
    <a href="javascript:di()">생성노드</a>
    </p>
</body>

    Onload¶

    <head>
    <script>
        function init() {
            p = document.getElementById("p");
            p.onmouseover = over;
            p.onmouseout = out;
        }

        function over() {
            p.style.backgroundColor = "pink";
        }

        function out() {
            p.style.backgroundColor = "white";
        }
    </script>
</head>
<body onload="init()">
    <p id ="p">색 변경</p>
</body>
